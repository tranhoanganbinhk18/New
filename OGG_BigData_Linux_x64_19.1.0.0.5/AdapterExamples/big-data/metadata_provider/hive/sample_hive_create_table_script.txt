CREATE SCHEMA GG_HIVE;

USE GG_HIVE;

CREATE EXTERNAL TABLE `TCUSTMER_HIVE`(
  `CUST_CODE2` VARCHAR(4),
  `NAME` VARCHAR(18),
  `CITY2` VARCHAR(11),
  `STATE` STRING);

CREATE EXTERNAL TABLE `TCUSTORD_HIVE`(
  `CUST_CODE2` VARCHAR(4),
  `ORDER_DATE2` TIMESTAMP,
  `PRODUCT_CODE` VARCHAR(8),
  `ORDER_ID` VARCHAR(50),
  `PRODUCT_PRICE` BIGINT,
  `PRODUCT_AMOUNT` BIGINT,
  `TRANSACTION_ID` VARCHAR(50));
